Q- Program to check if a given string is a palindrome using if/else statements and pattern matching.


import scala.io.StdIn;

object P4 {
  def isPalindrome(str: String): Boolean = {
    str match {
      case s if s.isEmpty || s.length == 1 => true
      case s if s.head == s.last => isPalindrome(s.substring(1, s.length - 1))
      case _ => false
    }
  }

  def main(args: Array[String]): Unit = {
    val input = StdIn.readLine("Enter a string: ")
    if (isPalindrome(input)) {
      println(s"'$input' is a palindrome.")
    } else {
      println(s"'$input' is not a palindrome.")
    }
  }
}
