Q- Program to calculate the average of all numbers between n1 and n2 (eg. 100 to 300.read values of n1 and n2 from user)


import scala.io.StdIn
object Q4{
    def main(args: Array[String]):Unit={
        try{
        print("Enter the First(Small) Number : ")
        val n1 = StdIn.readInt()
        print("Enter the Second(Big) Number : ")
        val n2 = StdIn.readInt()

        var sum = 0
        var count = 0
        for(i <- n1 to n2 ){
            count += 1
            sum += i
        }
        
        val avg = sum/count
        println(s"The Average between $n1 to $n2 = $avg")
        }catch{
            case e:ArithmeticException => println("Exception a number is not divisible by Zero")
        }
    }
}
