Q- Find the Factorial of given number using while loop 

import scala.io.StdIn
object Q3{
    def main(args:Array[String]):Unit={
        try{
            print("Enter the Number : ")
            val n = StdIn.readInt()

            var fact = 1
            var i = 1
            while(i <= n ){
                fact = fact * i
                i +=1
            }
            println(s"Factorial of $n = $fact")
        }catch{
            case e : Exception => print(e.getMessage)
        }
    }
}
