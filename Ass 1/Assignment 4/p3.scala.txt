Q- Program to reverse every word in a given string.


object p3 {
  def main(args: Array[String]): Unit = {
    println("Enter a sentence:")
    val sentence = scala.io.StdIn.readLine()

    val reversed = sentence
      .split(" ")
      .map(_.reverse)
      .mkString(" ")

    println(s"Reversed words: $reversed")
  }
}
