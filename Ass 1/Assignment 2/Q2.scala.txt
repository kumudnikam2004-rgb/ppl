Q- Program to calculate the sum of prime numbers between 1 to 100 .

object Q2{

    def isPrime(n : Int):Boolean={
        if(n <= 1)
            return false
        for(i <- 2 to n/2){
            if(n % i == 0) return false
        }
        true
    }

    def main(args: Array[String]):Unit={
        var sum = 0
        for(i <- 1 to 100){
            if(isPrime(i))
                sum += i
        }
        println("Sum of prime Numbers between 1 to 100 is = "+sum)
    }
}
