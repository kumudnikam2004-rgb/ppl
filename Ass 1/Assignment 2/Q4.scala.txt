Q- Program to print the Fibonacci series up to a given number using a user defined function.


import scala.io.StdIn

object Q4{

    def fibonacci(n: Int): Unit = {
        var a = 0
        var b = 1
        var count = 0

        while (count < n) {
          print(a + " ")
          val next = a + b
          a = b
          b = next
          count += 1
        }
    }


    def main(args : Array[String]):Unit={
        print("Enter the Number : ");
        val n = StdIn.readInt()

        print(s"Fibonacci serise of $n = ")
        fibonacci(n)
        println()
    }
}
